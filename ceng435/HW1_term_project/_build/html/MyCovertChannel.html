 16    def find_valid_port(self, base_port, port_range, target_bit, threshold, reverse_logic, two, max_attempts=100):
 17        """
 18        Find a port number that encodes the target bit correctly with randomness.
 19        Randomly sample port numbers in the total range up to max_attempts.
 20        """
 21        for _ in range(max_attempts):
 22            port = random.randint(base_port, base_port + port_range - 1)
 23            ones_count = bin(port).count('1')
 24            is_even = ones_count % two == 0
 25
 26            if reverse_logic:
 27                is_even = not is_even
 28
 29            if port > threshold:
 30                if (is_even and target_bit == '1') or (not is_even and target_bit == '0'):
 31                    return port
 32            else:
 33                if (not is_even and target_bit == '1') or (is_even and target_bit == '0'):
 34                    return port
 35
 36        return None  
 37
 38    def encode_bit(self, bit, bit_index, base_port, port_range, two, three, five, seven):
 39        threshold = base_port + (port_range // 2)
 40        reverse_logic = (
 41            (bit_index % two != 0) 
 42            ^ (bit_index % three == 0)  
 43            ^ ((bit_index * five) % seven > two)  
 44        )
 45
 46        port = self.find_valid_port(base_port, port_range, bit, threshold, reverse_logic, two)
 47        if port is None:
 48            raise ValueError("Could not find valid port for encoding")
 49
 50        return port

 52    def decode_bit(self, port, bit_index, base_port, port_range, two, three, five, seven):
 53        threshold = base_port + (port_range // 2)
 54        ones_count = bin(port).count('1')
 55        is_even = ones_count % two == 0
 56
 57        reverse_logic = (
 58            (bit_index % two != 0)  
 59            ^ (bit_index % three == 0)  
 60            ^ ((bit_index * five) % seven > two)  
 61        )
 62
 63        if reverse_logic:
 64            is_even = not is_even
 65
 66        if port > threshold:
 67            return '1' if is_even else '0'
 68        else:
 69            return '1' if not is_even else '0'

 71    def send(self, log_file_name, base_port, port_range, two, three, five, seven):
 72        """
 73        Send covert message by encoding bits in TCP destination ports.
 74        The message is sent as a series of packets, each encoding a single bit.
 75        Encoding is done using these steps:
 76        1. Determine the target bit to encode.
 77        2. Generate a random port number within the range.
 78        3. Check if the port number satisfies the parity condition for the target bit.
 79            Here is how the parity condition is determined:
 80            - If the port number is greater than the threshold, the parity should be even for '1' and odd for '0'.
 81            - If the port number is less than or equal to the threshold, the parity should be odd for '1' and even for '0'. 
 82            - Also to add an extra layer the parity condition is reversed based on the bit index and some randomness.
 83        4. If not, repeat the process until a valid port is found.
 84        5. Send the packet with the valid port number.
 85        """
 86        start_time = time.time()
 87        binary_message = self.generate_random_binary_message_with_logging(log_file_name)
 88
 89        for bit_index, bit in enumerate(binary_message):
 90            port = self.encode_bit(bit, bit_index, base_port, port_range, two, three, five, seven)
 91            packet = IP(dst="receiver")/TCP(dport=port)
 92            super().send(packet)
 93
 94        bits_per_second = len(binary_message) / (time.time() - start_time)
 95        print(f"Transmission rate: {bits_per_second} bits/second")
 96
 97    def receive(self, base_port, port_range, timeout, log_file_name, two, three, five, seven):
 98        """
 99        Receive and decode covert message from TCP destination ports.
100        The receiver listens for packets and decodes the bits from the destination ports.
101        Decoding is done using the reverse of the encoding process:
102        1. For each packet, extract the destination port number.
103        2. Determine the bit encoded in the port number using the parity condition.
104        3. Assemble the bits into bytes and decode the message.
105        4. Stop when the '.' character is received.
106        5. Log the received message to a file and print it.
107        """
108
109        received_chars = []
110        received_bits = []
111        current_byte = []
112        stop_sniffing = False
113
114        def process_packet(packet):
115            nonlocal stop_sniffing, received_chars, current_byte
116
117            if not packet.haslayer(TCP):
118                return False
119
120            dport = packet[TCP].dport
121
122            if base_port <= dport < base_port + port_range:
123                bit_index = len(received_bits)
124                bit = self.decode_bit(dport, bit_index, base_port, port_range, two, three, five, seven)
125                received_bits.append(bit)
126                current_byte.append(bit)
127
128                if len(current_byte) == 8:
129                    char = self.convert_eight_bits_to_character(''.join(current_byte))
130                    received_chars.append(char)
131                    current_byte = []
132
133                    if char == '.':
134                        stop_sniffing = True
135                        return True
136
137        sniff(
138            filter="tcp",
139            prn=process_packet,
140            timeout=timeout,
141            stop_filter=lambda _: stop_sniffing
142        )
143
144        received_text = ''.join(received_chars)
145
146        self.log_message(received_text, log_file_name)
147        print(f"Received message: {received_text}")
