31    def convert_string_message_to_binary(self, message):
32        """
33        - Converts the incoming string value to binary format.
34        - You do not have to use it directly; instead, you can use generate_random_binary_message or generate_random_binary_message_with_logging functions.
35        """
36        binary_message_to_transfer = ''.join(format(i, '08b') for i in bytearray(message, encoding='utf-8'))
37        return binary_message_to_transfer
38    def generate_random_message(self, min_length=5, max_length=10):
39        """
40        - You can use this function if you want to create a random string, e.g. for the payload of the packet.
41        """
42        assert 0 < min_length, "min_length must be bigger than 0"
43        assert min_length <= max_length, "min_length must be smaller than or equal to the max_length"
44        letters_digits = string.ascii_letters + string.digits
45        punctuation = ',?!'
46        all_chars = " " * 50 + letters_digits * 5 + punctuation
47        length = random.randint(min_length-1, max_length-1)
48        random_string = ''.join(random.choice(all_chars) for _ in range(length))
49        random_string += "."
50        return random_string
51    def generate_random_binary_message(self, min_length=50, max_length=100):
52        """
53        - It generates a random string whose length is between the min_length and max_length, and converts it to binary format.
54        - "." is the stopping character for the covert channel communication, so that it adds "." at the of the generated string without ruining the length restrictions.
55        - You must use this function to generate the message in binary format that will be transferred with covert channel communication.
56        """
57        random_message = self.generate_random_message(min_length=min_length, max_length=max_length)
58        random_binary_message = self.convert_string_message_to_binary(message=random_message)
59        return random_binary_message
60    def generate_random_binary_message_with_logging(self, log_file_name, min_length=50, max_length=100):
61        """
62        - Same as generate_random_binary_message() function with logging option.
63        """
64        random_message = self.generate_random_message(min_length=min_length, max_length=max_length)
65        random_binary_message = self.convert_string_message_to_binary(message=random_message)
66        self.log_message(message=random_message, log_file_name=log_file_name)
67        return random_binary_message
68    def sleep_random_time_ms(self, start=1, end=10):
69        """
70        - For the covert timing channels, you can use this function if you want to wait random time in miliseconds between start and end.
71        """
72        time.sleep(random.uniform(start, end) / 1000)
73    def convert_eight_bits_to_character(self, eight_bits):
74        """
75        - It can be used to convert the received eight bits to a character in the receiving operation.
